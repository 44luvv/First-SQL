create table if not exists teams (
	id serial primary key,
	team_name varchar(50)
);

create table if not exists singers (
	id serial primary key,
	full_name varchar(100),
	team_id integer references teams(id)
);

create table if not exists songs (
	id serial primary key,
	track_name varchar(20),
	singer_id integer references singers(id)
);

create table if not exists genres (
	id serial primary key,
	genre_name varchar(15)
);

create table if not exists singer_genres (
	singer_id integer references singers(id) on delete cascade,
	genre_id integer references genres(id) on delete cascade,
	primary key (singer_id, genre_id)
);

create table if not exists concerts (
	id serial primary key,
	title varchar(20),
	city varchar (20)
);

create table if not exists singer_concerts (
	singer_id integer references singers(id) on delete cascade,
	concert_id integer references concerts(id) on delete cascade,
	primary key (singer_id, concert_id)
);
